207                def send_request(data)
  
208                        if data[:method] == 'GET'
  
209                                response = HTTParty.get(API_URL + data[:url],
  
210                                  :query => data[:params],
  
211                                  :headers => { "PddToken" => PDD_TOKEN        }
  
212              )
  
213                        else
  
214                                response = HTTParty.post(API_URL + data[:url],
  
215                                  :query => data[:params],
  
216                                  :headers => { "PddToken" => PDD_TOKEN }
  
217                                )
  
218                        end
